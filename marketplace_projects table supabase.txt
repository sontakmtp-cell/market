create table public.marketplace_projects (
  id bigint generated always as identity not null,
  created_at timestamp with time zone not null default now(),
  updated_at timestamp with time zone null,
  client_user_id uuid null,
  title text not null,
  "shortDescription" text null,
  "fullDescription" text null,
  category text null,
  skills text[] null,
  "budgetMin" bigint null,
  "budgetMax" bigint null,
  currency text null default 'VND'::text,
  duration text null,
  deadline timestamp with time zone null,
  "isUrgent" boolean null default false,
  location text null,
  attachments jsonb[] null,
  objectives jsonb[] null,
  "technicalRequirements" jsonb[] null,
  deliverables jsonb[] null,
  client jsonb null,
  "proposalCount" integer null default 0,
  status text null default 'active'::text,
  constraint marketplace_projects_pkey primary key (id),
  constraint marketplace_projects_client_user_id_fkey foreign KEY (client_user_id) references auth.users (id) on delete CASCADE
) TABLESPACE pg_default;
