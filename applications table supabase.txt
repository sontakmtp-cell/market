create table public.applications (
  id bigint generated always as identity not null,
  created_at timestamp with time zone not null default now(),
  updated_at timestamp with time zone null,
  "jobId" bigint null,
  user_id uuid null,
  "fullName" text null,
  email text null,
  phone text null,
  address text null,
  linkedin text null,
  github text null,
  portfolio text null,
  summary text null,
  "technicalSkills" jsonb null,
  certifications text[] null,
  "experienceLevel" text null,
  cv jsonb null,
  "coverLetter" text null,
  "portfolioFile" jsonb null,
  certificates jsonb[] null,
  "customAnswers" jsonb null,
  "applicationDate" timestamp with time zone null default now(),
  status text null default 'submitted'::text,
  constraint applications_pkey primary key (id),
  constraint applications_jobId_fkey foreign KEY ("jobId") references recruitment_jobs (id) on delete CASCADE,
  constraint applications_user_id_fkey foreign KEY (user_id) references auth.users (id) on delete set null
) TABLESPACE pg_default;
